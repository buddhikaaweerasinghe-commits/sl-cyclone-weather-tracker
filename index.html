        function addLog(msg, type = 'SCAN') {

        }

            const list = document.getElementById('storm-list');

        function createStormMarker(storm) {

            const icon = L.divIcon({

                className: 'storm-marker',

                html: `<div class="pulse-ring" style="border-color:${storm.color}"></div><div class="storm-core" style="background:${storm.color}; box-shadow: 0 0 15px ${storm.color}"></div>`,

                iconSize: [40, 40]

            });

            return L.marker([storm.lat, storm.lon], { icon: icon }).bindPopup(`<b>${storm.name}</b><br>Wind: ${storm.wind} kts<br>Basin: ${storm.basin}`);

        }


        function refreshMap() {

            // Remove old markers

            Object.values(markers).forEach(m => map.removeLayer(m));

            markers = {};

            

            activeSystems.forEach(storm => {

                const m = createStormMarker(storm);

                m.addTo(map);

                markers[storm.id] = m;

            });

        }


        function focusStorm(id) {

            const storm = activeSystems.find(s => s.id === id);

            if(storm) {

                map.flyTo([storm.lat, storm.lon], 6);

                document.getElementById('focus-name').innerText = storm.name;

                addLog(`Focused view on ${storm.name}`, "NAV");

            }

        }
